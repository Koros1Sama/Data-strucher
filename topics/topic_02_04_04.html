DECLARE
  CURSOR emp_cursor (dept_no NUMBER) IS
    SELECT employee_id, last_name FROM employees
    WHERE department_id = dept_no;
  
  v_emp_rec emp_cursor%ROWTYPE;
BEGIN
  -- Open cursor for Department 10
  OPEN emp_cursor(10);
  LOOP
    FETCH emp_cursor INTO v_emp_rec;
    EXIT WHEN emp_cursor%NOTFOUND;
    DBMS_OUTPUT.PUT_LINE('Dept 10: ' || v_emp_rec.last_name);
  END LOOP;
  CLOSE emp_cursor;

  -- Reuse same cursor for Department 20
  OPEN emp_cursor(20);
  LOOP
    FETCH emp_cursor INTO v_emp_rec;
    EXIT WHEN emp_cursor%NOTFOUND;
    DBMS_OUTPUT.PUT_LINE('Dept 20: ' || v_emp_rec.last_name);
  END LOOP;
  CLOSE emp_cursor;
END;
