جملة SELECT INTO في PL/SQL

يمكنك استخدام أنواع SQL التالية داخل PL/SQL (Slide 25):

  • SELECT — جلب البيانات من الجداول
  • DML — INSERT, UPDATE, DELETE, MERGE
  • Transaction Control — COMMIT, ROLLBACK, SAVEPOINT

⚙️ SELECT INTO

الصيغة (Slide 26)

SELECT  select_list
INTO    {variable_name[, variable_name]...
        | record_name}
FROM    table
[WHERE  condition];

⚠️ قواعد مهمة:
• جملة INTO إجبارية في PL/SQL
• يجب أن يُرجع الاستعلام صفاً واحداً فقط

مثال 1: جلب قيمة واحدة (Slide 27)

SET SERVEROUTPUT ON
DECLARE
  fname VARCHAR2(25);
BEGIN
  SELECT first_name INTO fname
  FROM   employees
  WHERE  employee_id = 200;

  DBMS_OUTPUT.PUT_LINE('First Name is: ' || fname);
END;
/

مثال 2: جلب عدة قيم مع %TYPE (Slide 28)

DECLARE
  emp_hiredate  employees.hire_date%TYPE;
  emp_salary    employees.salary%TYPE;
BEGIN
  SELECT  hire_date, salary
  INTO    emp_hiredate, emp_salary
  FROM    employees
  WHERE   employee_id = 100;
END;
/

مثال 3: GROUP FUNCTION مع SELECT INTO (Slide 29)

SET SERVEROUTPUT ON
DECLARE
  sum_sal  NUMBER(10,2);
  deptno   NUMBER NOT NULL := 60;
BEGIN
  SELECT SUM(salary) INTO sum_sal
  FROM   employees
  WHERE  department_id = deptno;

  DBMS_OUTPUT.PUT_LINE('The sum of salary is ' || sum_sal);
END;
/

⚠️ مشكلة تشابه الأسماء | Naming Conventions (Slide 30)

-- ❌ مشكلة: أسماء المتغيرات = أسماء الأعمدة
DECLARE
  hire_date    employees.hire_date%TYPE;
  employee_id  employees.employee_id%TYPE := 176;
BEGIN
  SELECT hire_date
  INTO   hire_date        -- غموض!
  FROM   employees
  WHERE  employee_id = employee_id;  -- من أين يعرف Oracle أيهما؟
END;
/

💡 الحل: استخدم بادئة للمتغيرات مثل v_hire_date أو emp_hire_date

📝 نقاط هامة | Key Takeaways

  • INTO إجبارية في SELECT داخل PL/SQL
  • يجب إرجاع صف واحد فقط (وإلا: TOO_MANY_ROWS أو NO_DATA_FOUND)
  • يمكن استخدام %TYPE لضمان توافق الأنواع
  • يمكن استخدام Group Functions (SUM, COUNT, etc.)
  • تجنب تسمية المتغيرات بنفس أسماء الأعمدة
