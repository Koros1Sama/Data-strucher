🔲 الكتل المتداخلة | Nested Blocks

كتل PL/SQL يمكن تداخلها — قسم BEGIN...END أو EXCEPTION يمكن أن يحتوي على كتل داخلية.

مثال: Nested Blocks (Slide 16)

DECLARE
  outer_variable VARCHAR2(20) := 'GLOBAL VARIABLE';
BEGIN
  DECLARE
    inner_variable VARCHAR2(20) := 'LOCAL VARIABLE';
  BEGIN
    DBMS_OUTPUT.PUT_LINE(inner_variable);   -- ✅ يعمل
    DBMS_OUTPUT.PUT_LINE(outer_variable);   -- ✅ يعمل
  END;

  DBMS_OUTPUT.PUT_LINE(outer_variable);     -- ✅ يعمل
  -- DBMS_OUTPUT.PUT_LINE(inner_variable);  -- ❌ خطأ! غير مرئي هنا
END;
/

🔭 نطاق المتغيرات | Variable Scope & Visibility

مثال: نفس الاسم في كتلتين (Slide 17)

DECLARE
  father_name   VARCHAR2(20) := 'Patrick';
  date_of_birth DATE := '20-Apr-1972';
BEGIN
  DECLARE
    child_name    VARCHAR2(20) := 'Mike';
    date_of_birth DATE := '12-Dec-2002';   -- يحجب المتغير الخارجي!
  BEGIN
    DBMS_OUTPUT.PUT_LINE('Father: ' || father_name);
    DBMS_OUTPUT.PUT_LINE('DOB: ' || date_of_birth);    -- 12-Dec-2002 (الداخلي)
    DBMS_OUTPUT.PUT_LINE('Child: ' || child_name);
  END;

  DBMS_OUTPUT.PUT_LINE('DOB: ' || date_of_birth);      -- 20-Apr-1972 (الخارجي)
END;
/

حل المشكلة: Labels (Slide 18)

استخدم تسمية (Label) للوصول إلى المتغير الخارجي:

<>
DECLARE
  father_name   VARCHAR2(20) := 'Patrick';
  date_of_birth DATE := '20-Apr-1972';
BEGIN
  DECLARE
    child_name    VARCHAR2(20) := 'Mike';
    date_of_birth DATE := '12-Dec-2002';
  BEGIN
    -- الوصول لمتغير الكتلة الخارجية بالتسمية:
    DBMS_OUTPUT.PUT_LINE('Father DOB: ' || outer.date_of_birth);  -- 20-Apr-1972
    DBMS_OUTPUT.PUT_LINE('Child DOB: ' || date_of_birth);          -- 12-Dec-2002
  END;
END;
/
